is_mergeable_heap_selector

Synopsis 

namespace boost {

    template <typename Selector>
    struct is_mergeable_heap_selector
      : implementation_defined Boolean Integral Constant
    {
    };
}  // namespace boost

Description

A Boost.Heap model type is mergeable if and only if it defines a member function merge() that takes in a reference-to-non-const object of its own type and returns void. This metafunction evaluates to boost::mpl::true_ if the specified selector will cause container_gen to return a mergeable Boost.Heap model type; it evaluates to boost::mpl::false_ otherwise.

If BOOST_MPL_CFG_NO_HAS_XXX is defined, then you must specialize this metafunction to inherit from boost::mpl::true_ for your custom selector if and only if the type to be returned by container_gen is a mergeable Boost.Heap model type.

Where defined 

#include <boost/container_gen/is_mergeable_heap_selector.hpp>

Example

The boost::heap::heap_merge() function is the only way to merge two objects of different Boost.Heap model types or of the same Boost.Heap model type that is otherwise not mergeable. 

template <typename Heap>
void test_heap_merge(Heap& heap1, Heap& heap2, boost::mpl::false_)
{
    boost::heap::heap_merge(heap1, heap2);
}

However, the boost::heap::heap_merge() function may be less efficient than the merge() member function of mergeable Boost.Heap model types. 

template <typename Heap>
void test_heap_merge(Heap& heap1, Heap& heap2, boost::mpl::true_)
{
    heap1.merge(heap2);
}

Use this metafunction to choose the correct and most efficient merge implementation. 

template <typename Selector>
void test_heap_merge()
{
    typename boost::container_gen<Selector,int>::type heap_1, heap_2;
    typename boost::emplace_function_gen<Selector>::type pusher;

    pusher[heap_1](3)(5)(1);
    pusher[heap_2](2)(4)(0);
    test_heap_merge(
        heap_1
      , heap_2
      , boost::is_mergeable_heap_selector<Selector>()
    );

    for (BOOST_CHECK(heap_2.empty()); !heap_2.empty(); heap_2.pop())
    {
    }

    BOOST_CHECK(5 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(4 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(3 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(2 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(1 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(0 == heap_1.top());
    heap_1.pop();
    BOOST_CHECK(heap_1.empty());
}
